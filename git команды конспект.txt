git init - локальную папку превращает в репозиторий

git remote add origin https://github.com/artemkuvik/TestRepo.git привяжет

git status - где находимся, предложеие что делать

cd (путь к папке) - переход в папки

git add (файл) - добавить  изменения

git add -A - добавить всё

git commit - m 'коммит'- создать коммит

git push origin (куда(master)) - отправить на гитхаб

git clone ссылка - скопировать с git

git branch - посмотреть ветки

git fetch - скачать все ветки из git в локалку

git checkout (ветка) - переключаться межжду ветками

git checkout -b (название новой ветки) - создаёт новую ветку из той, в которой находишься в данный момент и сразу в неё переключается

git pull origin (ветка) - привезти изменения из git
